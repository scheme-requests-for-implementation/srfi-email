The following gem is by Duncan Smith...

(defun permute (ls)
  (if (null ls)
    nil
    (if (null (cdr ls))
      (list ls)
      (mapcon
       #'(lambda (x)
           (mapcar
            #'(lambda (y)
                (cons (car x) y))
            (permute (nconc (ldiff ls x) (cdr x)))))
       ls))))
